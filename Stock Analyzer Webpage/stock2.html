Original Pine Script (read-only):

//@version=6
strategy("MIDCAP MOMENTUM MASTER v6", shorttitle="MMM", overlay=true, max_bars_back=500,
         default_qty_type=strategy.percent_of_equity, default_qty_value=15,
         commission_type=strategy.commission.percent, commission_value=0.03, slippage=2)

// INPUT PARAMETERS
minPrice = input.float(50, "Minimum Price", minval=1, group="Universe Selection")
maxPrice = input.float(10000, "Maximum Price", minval=1, group="Universe Selection")
minVolume = input.int(500000, "Minimum Daily Volume", minval=10000, group="Universe Selection")

ema20_len = input.int(20, "20 EMA Length", group="Trend Analysis")
ema50_len = input.int(50, "50 EMA Length", group="Trend Analysis")
ema200_len = input.int(200, "200 EMA Length", group="Trend Analysis")

rsi_len = input.int(14, "RSI Length", group="Momentum")
rsi_low = input.float(40, "RSI Lower", minval=0, maxval=100, group="Momentum")
rsi_high = input.float(80, "RSI Upper", minval=0, maxval=100, group="Momentum")
vol_mult = input.float(1.5, "Volume Multiplier", minval=1.0, group="Momentum")

bb_len = input.int(20, "BB Length", group="Volatility")
bb_mult = input.float(2.0, "BB Multiplier", group="Volatility")
atr_len = input.int(14, "ATR Length", group="Volatility")

risk_pct = input.float(3.0, "Risk %", minval=0.5, maxval=10, group="Risk Management")
sl_pct = input.float(2.0, "Stop Loss %", minval=0.5, maxval=5, group="Risk Management")
tp_mult = input.float(3.0, "Take Profit Multiplier", minval=1.5, maxval=10, group="Risk Management")

// CALCULATIONS
ema20 = ta.ema(close, ema20_len)
ema50 = ta.ema(close, ema50_len)
ema200 = ta.ema(close, ema200_len)

rsi = ta.rsi(close, rsi_len)
[macd_line, signal_line, macd_hist] = ta.macd(close, 12, 26, 9)

[bb_upper, bb_basis, bb_lower] = ta.bb(close, bb_len, bb_mult)
atr = ta.atr(atr_len)

avg_vol = ta.sma(volume, 20)
vol_surge = volume > avg_vol * vol_mult

// FILTERS
universe_ok = close >= minPrice and close <= maxPrice and volume >= minVolume
trend_up = ema20 > ema50 and ema50 > ema200 and close > ema20
momentum_ok = rsi >= rsi_low and rsi <= rsi_high and macd_line > signal_line
volatility_ok = close > bb_basis

// STRATEGY
base_long = universe_ok and trend_up and momentum_ok and volatility_ok and vol_surge
breakout_long = universe_ok and breakout and vol_surge and rsi > 50 and rsi < 80
momentum_long = universe_ok and close > ema20 and close > ema50 and close > ema200 and rsi > 60

// ENTRY/EXIT
if long_condition and strategy.position_size == 0
    strategy.entry("LONG", strategy.long, qty=qty)

if strategy.position_size > 0
    strategy.exit("EXIT", "LONG", stop=sl_price, limit=tp_price)
    if close < ema20
        strategy.close("LONG", comment="EMA Exit")
